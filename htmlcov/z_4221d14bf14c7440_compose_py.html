109 def __init__(self, *args, event, **kwargs): 

110 # SubmissionFilterForm consumes ``event`` from kwargs, but MailTemplateForm 

111 # also needs it. Pre-set self.event so MailTemplateForm's attribute 

112 # fallback can find it after the kwarg is consumed. 

113 self.event = event 

114 super().__init__(*args, event=event, **kwargs) 

115 initial = kwargs.get("initial", {}) 

116 self.filter_search = initial.get("q") 

117 question = initial.get("question") 

118 if question: 

119 self.filter_question = ( 

120 self.event.questions.all().filter(pk=question).first() 

121 ) 

122 if self.filter_question: 

123 self.filter_option = self.filter_question.options.filter( 

124 pk=initial.get("answer__options") 

125 ).first() 

126 self.filter_answer = initial.get("answer") 

127 self.filter_unanswered = initial.get("unanswered") 

128 self.fields["submissions"].choices = [ 

129 (sub.code, sub.title) 

130 for sub in self.event.submissions.all().order_by("title") 

131 ] 

132 speakers_field = self.fields["speakers"] 

133 speakers_field.queryset = self.event.submitters.all().order_by("name") 

134 speakers_field.label_from_instance = lambda obj: obj.get_display_name() 

135 if len(self.event.locales) > 1: 

136 self.fields["subject"].help_text = _( 

137 "If you provide only one language, that language will be used for all emails. If you provide multiple languages, the best fit for each speaker will be used." 

138 ) 

139 self.warnings = []

141 @property 

142 def speaker_only_recipients(self): 

143 cleaned_data = getattr(self, "cleaned_data", None) 

144 if not cleaned_data or not cleaned_data.get("speakers"): 

145 return False 

146 has_submission_recipients = cleaned_data.get("submissions") or any( 

147 cleaned_data.get(key) for key in self.RECIPIENT_FILTER_FIELDS 

148 ) 

149 return not has_submission_recipients

151 def get_valid_placeholders(self, ignore_data=False): 

152 kwargs = ["event", "user", "submission", "slot"] 

153 if not self.event.current_schedule: 

154 kwargs.remove("slot") 

155 if not ignore_data and self.speaker_only_recipients: 

156 kwargs = [k for k in kwargs if k not in ("submission", "slot")] 

157 return get_available_placeholders(event=self.event, kwargs=kwargs)

159 def clean(self): 

160 cleaned_data = super().clean() 

161 has_filters = any(cleaned_data.get(key) for key in self.RECIPIENT_FILTER_FIELDS) 

162 added_submissions = cleaned_data.get("submissions") 

163 added_speakers = cleaned_data.get("speakers") 

164 

165 if not has_filters and not added_submissions and not added_speakers: 

166 raise forms.ValidationError( 

167 _( 

168 "Please select at least one filter or specific proposals/speakers as recipients." 

169 ) 

170 ) 

171 

172 if has_filters: 

173 submissions = ( 

174 self.filter_queryset(self.event.submissions) 

175 .select_related("track", "submission_type", "event") 

176 .with_sorted_speakers() 

177 ) 

178 else: 

179 submissions = self.event.submissions.none() 

180 

181 if added_submissions: 

182 specific_submissions = ( 

183 self.event.submissions.filter(code__in=added_submissions) 

184 .select_related("track", "submission_type", "event") 

185 .with_sorted_speakers() 

186 ) 

187 submissions = submissions | specific_submissions 

188 

189 result = [] 

190 for submission in submissions: 

191 slots = submission.current_slots or [] 

192 if slots: 

193 for slot in slots: 

194 result.extend( 

195 {"submission": submission, "slot": slot, "user": speaker.user} 

196 for speaker in submission.sorted_speakers 

197 ) 

198 else: 

199 result.extend( 

200 {"submission": submission, "user": speaker.user} 

201 for speaker in submission.sorted_speakers 

202 ) 

203 if added_speakers: 

204 result.extend({"user": speaker.user} for speaker in added_speakers) 

205 self._recipients = result 

206 return cleaned_data
