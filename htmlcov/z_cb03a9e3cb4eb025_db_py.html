10class Median(Aggregate): 

11 """Custom median aggregate that works with both PostgreSQL and SQLite.""" 

12 

13 function = "PERCENTILE_CONT" 

14 name = "median" 

15 output_field = FloatField() 

16 template = "%(function)s(0.5) WITHIN GROUP (ORDER BY %(expressions)s)" 

17 

18 def as_sqlite( 

19 self, compiler, connection, **extra_context 

20 ): # pragma: no cover -- vendor-specific SQLite fallback 

21 # SQLite doesn't have PERCENTILE_CONT, but has a median() extension function 

22 # However, it's not always available. We'll use a subquery approach instead. 

23 # For now, fall back to using the mean for SQLite 

24 self.function = "AVG" 

25 self.template = "%(function)s(%(expressions)s)" 

26 return super().as_sql(compiler, connection, **extra_context)

29class Translate(Transform): 

30 name = "translate" 

31 

32 _BASE_TEMPLATES = { 

33 "postgresql": ( 

34 "CASE " 

35 "WHEN %(expressions)s IS JSON OBJECT THEN " 

36 "COALESCE(" 

37 "NULLIF(%(expressions)s::json->>'{locale}', ''), " 

38 "%(expressions)s::json->>'en'," 

39 "(SELECT value FROM json_each_text(%(expressions)s::json) LIMIT 1)" 

40 ")" 

41 "ELSE %(expressions)s::text " 

42 "END" 

43 ), 

44 "sqlite": ( 

45 "CASE " 

46 "WHEN json_valid(%(expressions)s) THEN " 

47 "COALESCE(" 

48 "NULLIF(json_extract(%(expressions)s, '$.{locale}'), ''), " 

49 "json_extract(%(expressions)s, '$.en'), " 

50 "(SELECT value FROM json_each(%(expressions)s) WHERE json_each.type != 'object' LIMIT 1)" 

51 ")" 

52 "ELSE %(expressions)s " 

53 "END" 

54 ), 

55 } 

56 

57 def __init__(self, *args, **kwargs): 

58 super().__init__(*args, **kwargs) 

59 vendor = connection.vendor 

60 if vendor not in self._BASE_TEMPLATES: 

61 raise NotImplementedError(f"Translate not supported for {vendor}") 

62 self.base_template = self._BASE_TEMPLATES[vendor] 

63 

64 @property 

65 def template(self): 

66 # Lazy template eval in order to get the actual current language 

67 current_locale = translation.get_language() 

68 return self.base_template.format(locale=current_locale)
